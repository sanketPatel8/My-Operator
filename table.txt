CREATE TABLE shopify_orders (
  id BIGINT AUTO_INCREMENT PRIMARY KEY,        
  shopify_order_id BIGINT,                     
  order_number INT,                             
  confirmation_number VARCHAR(50),
  confirmed BOOLEAN,
  created_at DATETIME,
  processed_at DATETIME,
  currency VARCHAR(10),
  financial_status VARCHAR(50),
  fulfillment_status VARCHAR(50),
  subtotal_price DECIMAL(10,2),
  total_price DECIMAL(10,2),
  total_tax DECIMAL(10,2),
  total_discounts DECIMAL(10,2),
  total_shipping DECIMAL(10,2),
  name VARCHAR(50),                       
  order_status_url VARCHAR(212),
  phone VARCHAR(50),
  email VARCHAR(150),
  customer_id BIGINT,
  customer_name VARCHAR(150),
  customer_phone VARCHAR(50),
  payment_gateway VARCHAR(100),
  billing_address longtext,
  shipping_address longtext,
  customer longtext,
  line_items longtext,
  shipping_lines longtext,
  discount_codes longtext,
  raw_payload longtext,                             -- Full webhook JSON
  store_domain VARCHAR(150)                     -- Helpful if handling multiple stores
);
